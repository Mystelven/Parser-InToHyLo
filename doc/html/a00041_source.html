31 void SyntacticAnalyser::parseFormula(vector<Formula *> &formulas)
32 {
33  prepareParser();
34  int tk = lx->nextToken(), id = lx->currentValue;
35 
36  while(tk != END_FILE)
37  {
38  while(tk == COMMENT){lx->eatLine(); tk = lx->nextToken(); id = lx->currentValue;}
39 
40  if(tk == PROPOSITIONAL || tk == TRUE || tk == FALSE) // simple formula
41  {
42  switch(tk)
43  {
44  case PROPOSITIONAL : stackFormula.push_back(new PropositionalVariable(id)); break;
45  case TRUE : stackFormula.push_back(new Top()); break;
46  case FALSE : stackFormula.push_back(new Bottom()); break;
47  }
48  stackToken.push_back(FORMULA);
49  }
50  else if(tk == PAR_LEFT || tk == BEGIN || tk == PAR_RIGHT || tk == END) pushNull(tk);
51  else if(tk == BOX || tk == DIAMOND) pushUnary(tk, id);
52  else if(tk == NOT || tk == IMPLY || tk == EQUIV || tk == OR || tk == AND) pushUnary(tk);
53 
54  while(tk != COMMENT && reduce());
55 
56  if(tk == END) formulas.push_back(getFormula());
57 
58  tk = lx->nextToken();
59  id = lx->currentValue;
60  }
61 } // parseFormula

68 extern inline bool SyntacticAnalyser::reduce()
69 {
70 
71  if(stackToken.size() < 3) return false;
72 
73  // we always have BEGIN as first element
74  assert(stackToken[0] == BEGIN);
75  assert(stackToken.size() == stackFormula.size());
76 
77  int last = stackToken.size() - 1;
78 
79  if(stackToken[last] == FORMULA)
80  {
81  switch(stackToken[last - 1])
82  {
83  case BOX :
84  case DIAMOND :
85  case NOT :
86  stackFormula[last - 1]->addChild(stackFormula[last]);
87  stackToken[last - 1] = FORMULA;
88  stackFormula.pop_back(); stackToken.pop_back();
89  assert(stackToken.size() == stackFormula.size());
90  return true;
91  }
92  }
93 
94  else if(stackToken[last] == PAR_RIGHT || stackToken[last] == END)
95  {
96  if(stackToken[last - 1] == FORMULA && stackToken[last - 2] == PAR_LEFT) merge(PAR_LEFT, FORMULA, PAR_RIGHT);
97  else if(stackToken[last - 1] == FORMULA && stackToken[last - 2] == BEGIN) merge(BEGIN, FORMULA, END);
98  else
99  {
100  assert(stackToken.size() > 4);
101  assert(stackToken[last - 1] == FORMULA && stackToken[last - 3] == FORMULA);
102 
103  stackFormula[last - 2]->addChild(stackFormula[last - 3]);
104  stackFormula[last - 2]->addChild(stackFormula[last - 1]);
105 
106  stackFormula[last - 3] = stackFormula[last - 2];
107  stackFormula[last - 2] = stackFormula[last];
108  stackToken[last - 3] = FORMULA;
109  stackToken[last - 2] = stackToken[last];
110 
111  stackFormula.pop_back(); stackFormula.pop_back();
112  stackToken.pop_back(); stackToken.pop_back();
113  }
114  assert(stackToken.size() == stackFormula.size());
115  return true;
116  }
117 
118  else if(stackToken[last] == AND || stackToken[last] == OR || stackToken[last] == IMPLY || stackToken[last] == EQUIV)
119  {
120  if(stackToken[last - 2] == AND || stackToken[last - 2] == OR || stackToken[last - 2] == IMPLY || stackToken[last - 2] == EQUIV)
121  {
122  if(stackFormula[last]->priority <= stackFormula[last - 2]->priority)
123  {
124  assert(stackFormula.size() > 4);
125  stackFormula[last - 2]->addChild(stackFormula[last - 3]);
126  stackFormula[last - 2]->addChild(stackFormula[last - 1]);
127 
128  stackFormula[last - 3] = stackFormula[last - 2];
129  stackFormula[last - 2] = stackFormula[last];
130  stackToken[last - 3] = FORMULA;
131  stackToken[last - 2] = stackToken[last];
132 
133  stackFormula.pop_back(); stackFormula.pop_back();
134  stackToken.pop_back(); stackToken.pop_back();
135  assert(stackToken.size() == stackFormula.size());
136  return true;
137  }
138  }
139  }
140 
141  assert(stackToken.size() == stackFormula.size());
142  return false;
143 
144 } // reduce
